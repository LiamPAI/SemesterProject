22 /**
23 * @brief Class for computing parametric finite element stiffness matrices
24 * @details Implements stiffness matrix computation for second-order
25 * elements using either plane stress or plane strain formulation
26 */
27 class ParametricFEElementMatrix {
28 private:
29 //Declare our parameters for the calculation of the stiffness matrix, using Young's modulus, nu, and
30 //either the plane stress of plane strain matrix
31 const double youngmodulus{}; ///< Young's modulus (E)
32 const double nu{}; ///< Poisson's ratio (Î½)
33 Eigen::Matrix<double, 3, 3> D; ///< Constitutive matrix (plane stress/strain)
34
35 public:
36
37 //The default constructor is not allowed, as the parameters (Young's modulus, and nu) are required
38 //for the computation of the stiffness matrix
39 /** @brief Default constructor disabled - material parameters required */
40 ParametricFEElementMatrix() = delete;
41
42 //Constructor I intend to use
43 /**
44 * @brief Constructs element matrix calculator with material properties
45 * @param young Young's modulus
46 * @param nu Poisson's ratio
47 * @param planeStrain true for plane strain, false for plane stress
48 * @throws LFException if nu = 0.5 (plane strain) or nu = 1.0 (plane stress)
49 */
50 ParametricFEElementMatrix(double young, double nu, bool planeStrain) : youngmodulus {young}, nu {nu}
51 {
52 if (planeStrain) {
53 //D is our plane strain matrix, which we initialize here
54 LF_ASSERT_MSG(std::abs(nu - 0.5) >= (0.0001), "Value of nu (0.5) will lead to a divide by zero");
55 D << (1 - nu), nu, 0.0,
56 nu, (1- nu), 0.0,
57 0.0, 0.0, (1 - 2 * nu) / 2.0;
58 D *= youngmodulus / ((1 + nu) * (1 - 2 * nu));
59 }
60 else {
61 //Initialize D as a plane stress matrix
62 LF_ASSERT_MSG((std::abs(nu - 1.0) >= (0.0001)), "Value of nu (1.0) will lead to a divide by zero");
63 D << 1.0, nu, 0.0,
64 nu, 1.0, 0.0,
65 0.0, 0.0, (1 - nu) / 2.0;
66 D *= youngmodulus / (1 - nu * nu);
67 }
68 }
69
70 // The isActive and Eval function required to send to AssembleMatrixLocally later on
71 /**
72 * @brief Indicates if element should be processed
73 * @return Always returns true as all cells are integrated
74 */
75 bool isActive (const lf::mesh::Entity&) {return true;} //All cells will be integrated over
76
77 /**
78 * @brief Computes element stiffness matrix for given cell
79 * @details The returned matrix is of size 18x18 as a quadrilateral cell will have 9 nodes, so 18 entries due
80 * to the 2D nature of linear elasticity. For triangular cells, the final 6 rows and 6 columns are simply 0
81 * since second-order triangular cells have 6 nodes
82 *
83 * @param cell Mesh entity (element) to compute matrix for
84 * @return 18x18 element stiffness matrix for parametric elements
85 */
86 Eigen::Matrix<double, 18, 18> Eval(const lf::mesh::Entity &cell);
87
88 // This method will be used for post-processing, taking in the displacement vector, and return matrices
89 // for the stress and strains at the element's respective nodes
90 // This method is virtually identical to the one in linear_matrix computation
91 /**
92 * @brief Computes stress and strain at element nodes
93 * @param cell Mesh entity to compute for
94 * @param disp Global displacement vector
95 * @param dofh Degree of freedom handler
96 * @return Tuple of (stress matrix, strain matrix, nodal coordinates)
97 */
98 std::tuple<Eigen::MatrixXd, Eigen::MatrixXd, Eigen::MatrixXd> stressStrain(const lf::mesh::Entity &cell, Eigen::VectorXd &disp, const lf::assemble::DofHandler &dofh);
99
100 // The purpose of this method is to calculate the energy of a cell once we have obtained the displacement vector
101 // Note this method will use stressStrain as part of its calculation
102 /**
103 * @brief Calculates strain energy for an element
104 * @param cell Mesh entity to compute energy for
105 * @param disp Global displacement vector
106 * @param dofh Degree of freedom handler
107 * @return Strain energy of the element
108 */
109 double energyCalc(const lf::mesh::Entity &cell, Eigen::VectorXd &disp, const lf::assemble::DofHandler &dofh);
110 };

112 /**
113 * @brief Class for computing parametric finite element load vectors (of order 2)
114 * @details Handles computation of load vectors including body forces and surface tractions for second-order
115 * elements. This was added for completeness in the case that traction and body forces were to be implemented,
116 * but is not used within the implementation
117 */
118 class ParametricFELoadVector {
119 private:
120 //Body force, boundary flags, and a traction function will be required for the computation of the load vector
121 lf::mesh::utils::CodimMeshDataSet<bool> bd_flags; ///< Boundary flags
122 std::function<Eigen::Vector2d (const Eigen::Vector2d &)> traction_; ///< Surface traction function
123 std::function<Eigen::Vector2d (const Eigen::Vector2d &)> body_f; ///< Body force function
124
125 public:
126 //No default constructor, I want the bd_flags at least to be present when initialized
127 /** @brief Default constructor disabled - boundary flags required */
128 ParametricFELoadVector() = delete;
129
130 //The case where we don't have traction, 0 Neumann condition, the rhs vector will be 0
131 /**
132 * @brief Constructs with boundary flags only (zero loads)
133 * @param bd_flags Boundary condition flags
134 */
135 explicit ParametricFELoadVector(lf::mesh::utils::CodimMeshDataSet<bool> bd_flags)
136 : bd_flags(std::move(bd_flags)), traction_{}, body_f{} {}
137
138 //The case where there is traction but no body force
139 /**
140 * @brief Constructs with boundary flags and surface traction
141 * @param bd_flags Boundary condition flags
142 * @param traction Surface traction function
143 */
144 ParametricFELoadVector(lf::mesh::utils::CodimMeshDataSet<bool> bd_flags, std::function<Eigen::Vector2d (const Eigen::Vector2d &)> traction)
145 : bd_flags(std::move(bd_flags)), traction_(std::move(traction)), body_f{} {}
146
147 //Complete constructor, when there is all three
148 /**
149 * @brief Constructs with boundary flags, traction, and body forces
150 * @param bd_flags Boundary condition flags
151 * @param traction Surface traction function
152 * @param body Body force function
153 */
154 ParametricFELoadVector(lf::mesh::utils::CodimMeshDataSet<bool> bd_flags, std::function<Eigen::Vector2d (const Eigen::Vector2d &)> traction,
155 std::function<Eigen::Vector2d (const Eigen::Vector2d &)> body)
156 : bd_flags(std::move(bd_flags)), traction_(std::move(traction)), body_f(std::move(body)) {}
157
158
159 //Return the value at bd_flags for edges, and if there is no body force, don't loop over cells
160 /**
161 * @brief Determines if entity should be processed
162 * @param entity Mesh entity to check
163 * @return true if entity should be included in computation
164 * @throws LFException for invalid entity types
165 */
166 bool isActive (const lf::mesh::Entity &entity) {
167 const lf::base::RefEl ref_el (entity.RefEl());
168 switch (ref_el) {
169 case lf::base::RefEl::kSegment(): {
170 return bd_flags(entity);
171 }
172 case lf::base::RefEl::kTria(): case lf::base::RefEl::kQuad(): {
173 if (!body_f) {
174 return false;
175 }
176 return true;
177 }
178 default: {
179 LF_ASSERT_MSG(false, "Illegal cell type sent to isActive LinearFELoadVector");
180 }
181 }
182 }
183
184 //Declaration for Eval, this should be able to take both cells and edges, hence the name entity
185 //Has size up to 16 since we are now dealing with 8 nodes on a quadrilateral
186 /**
187 * @brief Computes load vector contribution for entity
188 * @details Similar to the Eval function for @link ParametricFEElementMatrix @endlink, the vector is of size 18 for
189 * quadrilateral elements. For trianglular cells, the last 6 entries will simply be zero
190 * @param entity Mesh entity (element or edge)
191 * @return 18-dimensional local load vector
192 */
193 Eigen::Vector<double, 18> Eval(const lf::mesh::Entity &entity);
194 };
195}
