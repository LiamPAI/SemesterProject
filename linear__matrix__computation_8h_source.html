16  * @brief Class for computing linear finite element stiffness matrices
17  * @details Implements stiffness matrix computation for linear elastic
18  * problems using either plane stress or plane strain formulation
19  */
20  class LinearFEElementMatrix {
21  private:
22  //Young's modulus, nu, and D are parameters we require for the computation of the stiffness matrix
23  //D is either the plane-strain matrix or plane-stress matrix depending on the mesh
24  const double youngmodulus {}; ///< Young's modulus (E)
25  const double nu {}; ///< Poisson's ratio (ν)
26  Eigen::Matrix<double, 3, 3> D; ///< Constitutive matrix (plane stress/strain)
27 
28  public:
29  //The default constructor is not allowed, as the parameters (Young's modulus, and nu) are required
30  //for the computation of the stiffness matrix
31  /** @brief Default constructor disabled - material parameters required */
32  LinearFEElementMatrix() = delete;
33 
34  //Constructor I intend to use
35  /**
36  * @brief Constructs element matrix calculator with material properties
37  * @param young Young's modulus
38  * @param nu Poisson's ratio
39  * @param planeStrain true for plane strain, false for plane stress
40  * @throws LFException if nu = 0.5 (plane strain) or nu = 1.0 (plane stress)
41  */
42  LinearFEElementMatrix(double young, double nu, bool planeStrain) : youngmodulus {young}, nu {nu}
43  {
44  if (planeStrain) {
45  //D is our plane strain matrix, which we initialize here
46  LF_ASSERT_MSG(std::abs(nu - 0.5) >= (0.0001), "Value of nu (0.5) will lead to a divide by zero");
47  D << (1 - nu), nu, 0.0,
48  nu, (1- nu), 0.0,
49  0.0, 0.0, (1 - 2 * nu) / 2.0;
50  D *= youngmodulus / ((1 + nu) * (1 - 2 * nu));
51  }
52  else
53  //Initialize D as a plane stress matrix
54  LF_ASSERT_MSG((std::abs(nu - 1.0) >= (0.0001)), "Value of nu (1.0) will lead to a divide by zero");
55  D << 1.0, nu, 0.0,
56  nu, 1.0, 0.0,
57  0.0, 0.0, (1 - nu) / 2.0;
58  D *= youngmodulus / (1 - nu * nu); {
59  }
60  }

149  //Additionally, I have not included the possibility that we compute the traction in just one direction instead of both
150  /**
151  * @brief Determines if entity should be processed
152  * @param entity Mesh entity to check
153  * @return true if entity should be included in computation
154  * @throws LFException for invalid entity types
155  */
156  bool isActive (const lf::mesh::Entity &entity) {
157  const lf::base::RefEl ref_el (entity.RefEl());
158  switch (ref_el) {
159  case lf::base::RefEl::kSegment(): {
160  return bd_flags(entity);
161  }
162  case lf::base::RefEl::kTria(): case lf::base::RefEl::kQuad(): {
163  if (!body_f) {
164  return false;
165  }
166  return true;
167  }
168  default: {
169  LF_ASSERT_MSG(false, "Illegal cell type sent to isActive LinearFELoadVector");
170  }
171  }
172  }
