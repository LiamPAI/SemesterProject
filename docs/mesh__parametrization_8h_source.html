24// This file will contain the structure of the primary class I will use to hold the lower dimensional mesh
25// The class will consist of terminal points (3 per "branch"), a width for each branch, and a vector for the direction of each terminal point
26// It will also contain helper functions to ensure that a given parametrization is valid
27// e.g. branches don't overlap anywhere except for at the center, the branches share points at the center

30/**
31 * @brief Parameters for finite element calculations and mesh generation
32 * @details Contains material properties and mesh configuration parameters to be used for calculations with
33 * individual parametrizations and the overall mesh
34 */
35struct calculationParams {
36 double yieldStrength; ///< Material yield strength
37 double youngsModulus; ///< Young's modulus (E)
38 double poissonRatio; ///< Poisson's ratio (ν)
39 double meshSize; ///< Mesh element size parameter
40 int order; ///< Finite element order (1 or 2)
41
42 /**
43 * @brief Constructs calculation parameters
44 * @param yStren Yield strength
45 * @param E Young's modulus
46 * @param nu Poisson's ratio
47 * @param ms Mesh size
48 * @param o Element order (must be 1 or 2)
49 * @throws LFException if order is not 1 or 2
50 */
51 calculationParams(double yStren, double E, double nu, double ms, int o)
52 : yieldStrength(yStren), youngsModulus(E), poissonRatio(nu), meshSize(ms), order(o)
53 {
54 LF_ASSERT_MSG(o == 1 or o == 2, "Value of order must be 1 or 2, but is instead " << o);
55 }
56};

58/**
59 * @brief Container for mesh parametrization geometric data
60 * @details Stores branch information including widths, terminal points, and direction vectors
61 * in a locally ordered format
62 *
63 * Matrix Ordering Convention:
64 * For a parametrization with n branches, widths will have shape n by 3, each row representing a new branch,
65 * and each column corresponding to the same columns for the terminals and the vectors
66 * Both vectors and terminals will have shape 2n by 3, given the 2D nature of the terminal points and vectors
67 *
68 * Local Ordering Convention:
69 * For multi-branch parametrizations, column 0 contains the points related to the "center" shape, such as a triangle
70 * a 3 branch multi parametrization
71 */
72struct MeshParametrizationData {
73 int numBranches{}; ///< Number of branches in parametrization
74 Eigen::MatrixXd widths; ///< Matrix of branch widths (numBranches × 3)
75 Eigen::MatrixXd terminals; ///< Matrix of terminal points (2*numBranches × 3)
76 Eigen::MatrixXd vectors; ///< Matrix of direction vectors (2*numBranches × 3)
77
78 /** @brief Default constructor */
79 MeshParametrizationData() = default;
80
81 /**
82 * @brief Constructs mesh parametrization with brief validation
83 * @param num Number of branches
84 * @param w Width matrix
85 * @param t Terminal points matrix
86 * @param v Direction vectors matrix
87 * @throws LFException if matrix dimensions don't match branch count
88 */
89 MeshParametrizationData(int num, Eigen::MatrixXd w, Eigen::MatrixXd t, Eigen::MatrixXd v)
90 : numBranches(num), widths(std::move(w)), terminals(std::move(t)), vectors(std::move(v)) {
91 LF_ASSERT_MSG(num == widths.rows() and widths.cols() == 3,
92 "Number of branches is not equal to the amount of _widths given in MeshParam");
93 LF_ASSERT_MSG(terminals.rows() / 2 == num and terminals.cols() == 3,
94 "Size of _terminals is invalid in MeshParam");
95 LF_ASSERT_MSG(vectors.rows() / 2 == num and vectors.cols() == 3, "Size of _vectors is invalid in MeshParam");
96 }
97};
