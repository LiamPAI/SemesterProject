11// The purpose of this class is to map points from one line to another line, which serves to provide the displacement
12// boundary conditions on a mesh parametrization

14/**
15 * @brief Class for mapping points between two line segments
16 * @details Provides functionality to map points from one line segment to another,
17 * find if lines intersect, and analyze geometric relationships between
18 * the segments
19 */
20class LineMapping {
21private:
22 Eigen::Vector2d leftStart; ///< Start point of first line segment
23 Eigen::Vector2d leftEnd; ///< End point of first line segment

26 Eigen::Matrix3d transformationMatrix; ///< Matrix for point transformation

28 /**
29 * @brief Computes the transformation matrix for point mapping

31 */
32 void computeTransformationMatrix();

34 /**
35 * @brief Calculates the shortest distance from a point to a line segment
36 * @param point Point to check
37 * @param lineStart Start point of line segment
38 * @param lineEnd End point of line segment
39 * @return Shortest distance from point to line segment
40 */
41 double distanceToLine(const Eigen::Vector2d& point, const Eigen::Vector2d& lineStart, const Eigen::Vector2d& lineEnd) const;

43public:
44 /**
45 * @brief Constructs a line mapping between two line segments

76 bool isPointOnFirstLine(const Eigen::Vector2d& point, double tolerance = 1e-6) const;

78 /**
