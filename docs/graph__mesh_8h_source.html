14// The purpose of the following data structure is to store the planned mesh, which is a 2D graph-like mesh, where
15// both the edges and the nodes are 2D. The naming system required for this to work is that all "nodes" and "edges"
16// are labeled, such that the surrounding curves (or boundary) are labeled "NB0, NB1, .." and "EB0, EB1,...",
17// respectively. Additionally, the physical 2D surface as well must be labeled, using the same number as the
18// corresponding boundary name. The below functions will also provide some of the training parameters for the NN,
19// as well as the compatibility conditions when running the "FEM" calculation on the NN

21// This struct is used to give an ordering on the compatibility conditions when optimizing the trained NN to output
22// the result from an FEM calculation
23/**
24 * @brief Container for mesh part compatibility conditions
25 * @details Stores information about how different parts of a split mesh relate to each other by providing indices
26 * for which displacement vectors, points, or terminals must be equal
27 */
28struct CompatibilityCondition {
29 std::pair<int, int> indices; ///< Indices of connected parts (first, second)
30 std::pair<int, int> firstLocation; ///< Location in first part (branch number, side)
31 std::pair<int, int> secondLocation; ///< Location in second part (branch number, side)
32
33 /**
34 * @brief Constructs compatibility condition between mesh parts
35 * @param inds Part indices pair
36 * @param firstLoc Location in first part
37 * @param secondLoc Location in second part
38 */
39 CompatibilityCondition(std::pair<int, int> inds, std::pair<int, int> firstLoc, std::pair<int, int> secondLoc)
40 : indices(std::move(inds)),
41 firstLocation(std::move(firstLoc)),
42 secondLocation(std::move(secondLoc)) {}
43};

67// The two structs below are used when generating the compatibility conditions to ensure each condition is unique
68/**
69 * @brief Hash function for integer pairs
70 * @details Provides unique hashing for unordered pairs of integers to store compatibility conditions
71 */
72struct IntPairHash {
73 std::size_t operator()(const std::pair<int, int> &p) const {
74 int a = std::min(p.first, p.second);
75 int b = std::max(p.first, p.second);
76 return ((a + b) * (a + b + 1) / 2) + b;
77 }
78};
79
80/**
81 * @brief Equality comparison for integer pairs
82 * @details Treats pairs as unordered (a,b) equals (b,a), useful when storing the compatibility conditions uniquely
83 */
84struct IntPairEqual {
85 bool operator()(const std::pair<int, int> &left, const std::pair<int, int> &right) const {
86 return (left.first == right.first and left.second == right.second) or
87 (left.first == right.second and left.second == right.first);
88 }
89};
