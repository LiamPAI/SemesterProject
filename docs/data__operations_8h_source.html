33// Overload the () operator so that we have a hash function for std::vector<int>,
34// useful when creating the point data set so we don't repeat flipped vector entries in the parametrization dataset
35/**
36* @brief Hash functor for vectors of integers
37* @details Useful for dealing with tags in the @link ParametrizationEntry @endlink and @link PointEntry @endlink
38*/
39struct VectorHash {
40 /**
41 * @brief Compute hash value for a vector of integers
42 * @param v Vector to be hashed
43 * @return size_t Hash value of the vector
44 */
45 size_t operator()(const std::vector<int>& v) const {
46 std::hash<int> hasher;
47 size_t seed = 0;
48 for (int i : v) {
49 seed ^= hasher(i) + 0x9e3779b9 + (seed << 6) + (seed >> 2); // "Fowler-Noll-Vo" hash function
50 }
51 return seed;
52 }
53};

69// Another struct that contains the exact same information as MeshParametrization butter, but only the 6 points of
70// each branch, to see how well the NN performs on both
71/**
72* @brief Container for parametrization of type points
73* @details The points matrix follows a strict local ordering, specifically for parametrizations with 3+ branches
74*
75* Matrix Structure:
76* For a parametrization with n branches, the points matrix has dimensions (4n * 3):
77* - 4n rows: the top 2 rows all represent one side of the parametrization, the bottom rows represent the points
78* on the other side
79* - 3 columns: For the 3 parts of the parametrization that we are taking points from, where the middle column
80* represents the "middle" points of the parametrization
81*
82* Local Ordering Convention:
83* For multi-branch parametrizations, column 0 contains the points related to the "center" shape, such as a triangle
84* a 3 branch multi parametrization
85* - Top 4 rows represent the 1st branch, the second 4 rows represent the 2nd branch, and so on
86*/
87struct ParametrizationPoints {
88 int numBranches; ///< Number of branches in the parametrization(1, 3+)
89 Eigen::MatrixXd points; ///< Matrix containing the locally-ordered point data
90
91 /**
92 * @brief Constructs parametrization points for a given number of branches
93 * @param num Number of branches
94 * @param p Matrix of points for this parametrization (must have 4 * num rows)
95 * @throws LFException if matrix dimensions don't match the branch count
96 */
97 ParametrizationPoints(int num, Eigen::MatrixXd p) : numBranches(num), points(std::move(p)) {
98 LF_ASSERT_MSG(numBranches == points.rows() / 4,
99 "Number of branches is not equal to the correct number of rows when initializing "
100 "ParametrizationPoints, num is " << numBranches <<
101 " and points has " << points.rows() << " rows");
102 }
103};

105// Instead of the parametrization, the actual polynomial points represented by the parametrization are used here, this
106// is for the purpose of testing both methods to see which works better when training the NN
107/**
108* @brief Container for a complete data entry containing @link ParametrizationPoints @endlink
109* @details Stores a data entry of type @link ParametrizationPoints @endlink along with the displacement vectors,
110* energy difference, and the vector of tags
111*
112* Local Ordering Convention:
113* - Displacements: (4 x 2) for single-branch parametrizations, and (4n x 1) for multi-branch parametrizations
114*
115* -Single-branch parametrizations:
116* Given the 4 x 3 structure of the equivalent points matrix the left column of displacements will correspond to the
117* left points in the points matrix, and vice versa for the right side
118* -Multi-branch parametrizations:
119* Given that the left-side points of the points matrix correspond to the "center" points of the parametrization, the
120* displacements matrix corresponds to the right side of the points matrix, with each point being matched up directly,
121* as they contain the same amount of rows
122*/
123struct PointEntry {
124 int numBranches; ///< Number of branches in the parametrization
125 ParametrizationPoints points; ///< Points containing the geometry of the parametrization
126 Eigen::MatrixXd displacements; ///< Matrix of displacement vectors as boundary conditions
127 double energyDifference; ///< Energy difference of the FEM problem
128 std::vector<int> tags; ///< Vector of tags to identify the data entry
129
130 /**
131 * @brief Constructs a complete parametrization entry
132 * @param num Number of branches
133 * @param p Parametrization containing points
134 * @param disp Displacement vectors as boundary conditions
135 * @param eD Energy difference
136 * @param tags Vector of classification tags
137 */
138 PointEntry(int num, ParametrizationPoints p, Eigen::MatrixXd disp, double eD, std::vector<int> tags)
139 : numBranches(num), points(std::move(p)), displacements(std::move(disp)), energyDifference(eD),
140 tags(std::move(tags)) {}
141};
